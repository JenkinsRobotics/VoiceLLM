2) New: core/metrics.py

from dataclasses import dataclass, field
from time import perf_counter
import csv, os, math

@dataclass
class TurnMetrics:
    wake_ts: float = 0.0
    listen_start_ts: float = 0.0
    listen_end_ts: float = 0.0
    stt_text: str = ""
    llm_first_token_ts: float = 0.0
    llm_done_ts: float = 0.0
    tts_start_ts: float = 0.0
    tts_end_ts: float = 0.0
    tokens: int = 0

    def as_row(self):
        def d(a,b): return round((b-a)*1000) if a and b else None
        return {
          "wake→listen(ms)": d(self.wake_ts, self.listen_start_ts),
          "listen dur(ms)": d(self.listen_start_ts, self.listen_end_ts),
          "stt→1st_token(ms)": d(self.listen_end_ts, self.llm_first_token_ts),
          "1st→tts_start(ms)": d(self.llm_first_token_ts, self.tts_start_ts),
          "tts dur(ms)": d(self.tts_start_ts, self.tts_end_ts),
          "e2e wake→tts_end(ms)": d(self.wake_ts, self.tts_end_ts),
          "tokens": self.tokens,
          "tok/s": round(self.tokens / max(1e-6, (self.llm_done_ts - self.llm_first_token_ts)), 2) if self.llm_first_token_ts and self.llm_done_ts else None,
          "stt_text": self.stt_text[:160]
        }

class MetricsLog:
    def __init__(self, path="metrics.csv"):
        self.path = path
        self._init()

    def _init(self):
        if not os.path.exists(self.path):
            with open(self.path, "w", newline="") as f:
                w = csv.DictWriter(f, fieldnames=list(TurnMetrics().as_row().keys()))
                w.writeheader()

    def write(self, tm: TurnMetrics):
        with open(self.path, "a", newline="") as f:
            w = csv.DictWriter(f, fieldnames=list(tm.as_row().keys()))
            w.writerow(tm.as_row())

def now(): return perf_counter()

3) New: audio/aec.py

Option A: pyaec (default)

import numpy as np
from pyaec import AEC  # SpeexDSP-based

class AECWrapper:
    """
    Near-end: mic input (what we capture)
    Far-end:  playback/reference (what we are sending to speakers)
    """
    def __init__(self, sample_rate=16000, frame_samples=160):
        # frame_samples: 10 ms at 16 kHz = 160 samples
        self.sr = sample_rate
        self.frame = frame_samples
        self.ec = AEC(sample_rate=self.sr, frame_size=self.frame)  # uses sane defaults

    def process(self, near_int16: np.ndarray, far_int16: np.ndarray | None):
        # Split into 10ms frames; if far is None, pass zeros.
        if far_int16 is None: far_int16 = np.zeros_like(near_int16)
        out = np.empty_like(near_int16)
        for i in range(0, len(near_int16), self.frame):
            n = near_int16[i:i+self.frame]
            f = far_int16[i:i+self.frame]
            if len(n) < self.frame:
                padn = np.pad(n, (0, self.frame-len(n)), mode='constant')
                padf = np.pad(f, (0, self.frame-len(f)), mode='constant')
                y = self.ec.process(padn, padf)[:len(n)]
            else:
                y = self.ec.process(n, f)
            out[i:i+len(y)] = y
        return out

class RingAudio:
    # ... existing ...
    def push_far(self, frames: np.ndarray):
        # maintain a parallel far-end ring if you want; simplest: reuse buf or keep another
        if not hasattr(self, "far"):
            from collections import deque
            self.far = deque(maxlen=self.maxlen)
        self.far.extend(frames.astype(np.int16).tolist())

    def snapshot_far(self):
        import numpy as np
        if not hasattr(self, "far"): return np.zeros(0, dtype=np.int16)
        return np.array(self.far, dtype=np.int16)

6) Patch: orchestrator/orchestrator.py (wire AEC + metrics)

from core.metrics import MetricsLog, TurnMetrics, now
from audio.aec import AECWrapper
# ...

class Orchestrator:
    def __init__(self, bus, llm, tts):
        # ...
        self.metrics = MetricsLog()
        self.cur = None
        self.aec = AECWrapper(sample_rate=SAMPLE_RATE, frame_samples=160)  # 10ms
        # Hook TTS playback to provide far-end to AEC ring
        tts.on_playback_chunk = lambda chunk: self.ring.push_far(chunk)
        self._bind()

    def _bind(self):
        def on_bus():
            while True:
                msg = self.bus.get()
                if msg.topic == "stt.text":
                    if self.cur: self.cur.listen_end_ts = now(); self.cur.stt_text = msg.payload
                    self.state.set(SysState.THINKING)
                    threading.Thread(target=self.llm.ask_stream, args=(msg.payload,), daemon=True).start()

                elif msg.topic == "llm.token":
                    if self.cur and not self.cur.llm_first_token_ts:
                        self.cur.llm_first_token_ts = now()
                        if not self.cur.tts_start_ts:
                            self.cur.tts_start_ts = self.cur.llm_first_token_ts
                    self.cur.tokens += 1
                    if self.state.value != SysState.RESPONDING:
                        self.state.set(SysState.RESPONDING)
                    self.tts.feed_text(msg.payload)

                elif msg.topic == "llm.done":
                    if self.cur: self.cur.llm_done_ts = now()
                    self.state.set(SysState.IDLE)

                elif msg.topic == "tts.chunk_done":
                    pass

                elif msg.topic == "tts.done":
                    if self.cur:
                        self.cur.tts_end_ts = now()
                        self.metrics.write(self.cur)
                        self.cur = None

        threading.Thread(target=on_bus, daemon=True).start()

        def on_audio(frames):
            # AEC: align near/far roughly by using latest far snapshot
            far = self.ring.snapshot_far()
            # match lengths (simple last-N alignment)
            if len(far) >= len(frames): far_slice = far[-len(frames):]
            else:
                import numpy as np
                pad = np.zeros(len(frames) - len(far), dtype=np.int16)
                far_slice = np.concatenate([pad, far]) if len(far) else pad
            clean = self.aec.process(frames, far_slice)

            self.ring.push(clean)  # store cleaned audio

            if self.state.value == SysState.IDLE:
                kw, score = self.wake.score(self.ring.snapshot())
                if score > 0.6:
                    self.state.set(SysState.LISTENING)
                    self.stt.reset()
                    self.cur = TurnMetrics()
                    self.cur.wake_ts = now()
                    self.cur.listen_start_ts = self.cur.wake_ts
            elif self.state.value == SysState.LISTENING:
                self.stt.feed(clean)

        self.stream = start_input(on_audio, blocksize=160)  # 10ms @16k

Notes:
	•	We emit to CSV at metrics.csv in the project root after each turn.
	•	We grabbed 10 ms blocks to keep AEC happy and reduce latency.
	•	The far‑end alignment is a pragmatic “last‑N” approach; you can refine with a small FIFO to match your playback path exactly.
